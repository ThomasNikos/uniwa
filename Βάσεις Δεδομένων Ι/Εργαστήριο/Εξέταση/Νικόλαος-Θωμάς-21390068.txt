-- Ερώτημα 1.

SELECT JOB AS "Επάγγελμα",
AVG(SAL) AS "Μ.Ο. Μισθού",
SUM(SAL+IFNULL(COMM,0)) AS "Σύνολο Αποδοχών",
COUNT(EMPNO) AS "Σύνολο Εργαζομένων"
FROM EMP
GROUP BY JOB;

-- Ερώτημα 2.

SELECT JOB AS "Επάγγελμα",
AVG(SAL) AS "Μ.Ο. Μισθού",
SUM(SAL+IFNULL(COMM,0)) AS "Σύνολο Αποδοχών",
COUNT(EMPNO) AS "Σύνολο Εργαζομένων"
FROM EMP
GROUP BY JOB
HAVING COUNT(EMPNO)>= 2;
