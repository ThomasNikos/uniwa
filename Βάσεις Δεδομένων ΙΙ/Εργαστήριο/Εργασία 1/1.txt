SELECT E.EMPNO, E.NAME, J.JOB_DESCR, J.SAL, E.DEPTNO
FROM EMP E, JOB J
WHERE E.JOBNO = "100"
		AND J.JOBCODE = "100";	
		
SELECT MAX(J.SAl) AS MAXSAL, MIN(J.SAl) AS MINSAL, AVG(J.SAl) AS AVGSAL, COUNT(J.SAl) AS GOTSAL, COUNT(COMM) AS GOTCOMM, COUNT(*) AS NUMOFEMP
FROM JOB J, EMP E
WHERE J.JOBCODE=E.JOBNO;

SELECT MAX(J.SAL) AS MAXSAL, AVG(J.SAL) AS AVGSAL
FROM JOB J
WHERE J.JOB_DESCR = "ΠΩΛΗΤΗΣ";

SELECT E.EMPNO, E.NAME, J.JOB_DESCR, J.SAL, E.DEPTNO
FROM EMP E, JOB J
WHERE E.JOBNO = "200"
	AND J.JOBCODE = "200"
	AND J.SAL BETWEEN 1000 AND 2500;
	
SELECT E.EMPNO, E.NAME, J.JOB_DESCR, J.SAL, E.DEPTNO
FROM EMP E, JOB J
WHERE NAME LIKE "%Ρ%"
	AND J.JOBCODE = E.JOBNO;
	
USE PERSONNEL;
SELECT E.EMPNO, E.NAME, J.JOB_DESCR, J.SAL, E.DEPTNO
FROM EMP E, JOB J
WHERE E.JOBNO = J.JOBCODE
ORDER BY E.DEPTNO, J.SAL;

USE PERSONNEL;
SELECT
(SELECT COUNT(E.EMPNO)
FROM EMP E, DEPT D
WHERE E.DEPTNO = "50"
	AND D.DEPTNO = E.DEPTNO) AS COUNTEMP_50,
(SELECT AVG(J.SAL)
FROM EMP E, JOB J, DEPT D
WHERE E.DEPTNO = "50"
	AND D.DEPTNO = E.DEPTNO
	AND E.JOBNO = J.JOBCODE) AS AVGSAL_50,
(SELECT COUNT(E.EMPNO)
FROM EMP E, DEPT D
WHERE E.DEPTNO = "60"
	AND D.DEPTNO = E.DEPTNO) AS COUNTEMP_60,
(SELECT AVG(J.SAL)
FROM EMP E, JOB J, DEPT D
WHERE E.DEPTNO = "60"
	AND D.DEPTNO = E.DEPTNO
	AND E.JOBNO = J.JOBCODE) AS AVGSAL_60,
(SELECT COUNT(E.EMPNO)
FROM EMP E, DEPT D
WHERE E.DEPTNO = "70"
	AND D.DEPTNO = E.DEPTNO) AS COUNTEMP_70,
(SELECT AVG(J.SAL)
FROM EMP E, JOB J, DEPT D
WHERE E.DEPTNO = "70"
	AND D.DEPTNO = E.DEPTNO
	AND E.JOBNO = J.JOBCODE) AS AVGSAL_70;
